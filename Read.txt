package voc;
import java.util.*;
import java.io.*;
/*
 * 英文測驗程式
 * 可測驗格式正確的英文單字
 */

//輸入:格式正確的英文單字(txt檔)
//輸出:測驗程式
//last:Write.java
//next:null

public class Read {
	
	static Scanner cin = new Scanner(System.in);
	
	public static void main(String[] args) throws FileNotFoundException {
		
		File file =new File("C:\\Users\\user\\Desktop\\Voc\\L7.txt");
		Scanner cout = new Scanner(new FileInputStream(file));
		
		Word [] word = new Word [1001] ;
		
		int vocabulary=0;
		
		while(cout.hasNext()) {
			String str=cout.nextLine();
			if(str.equals("")) {
				continue;
			}
			word[vocabulary] = new Word();
			
			int split1,split2;
			for(int i=0 ; true ; i++) {
				System.out.print(str.charAt(i));
				if(str.charAt(i)==' ') {
					split1=i;
					break;
				}
			}
			System.out.println();
			for(int i=split1+1 ; true ; i++) {//error split space format error
				System.out.print(str.charAt(i));
				if(str.charAt(i)==' ') {
					split2=i;
					break;
				}
			}
			
			word[vocabulary].ENG=str.substring(0 , split1);
			
			word[vocabulary].CH=str.substring(split2);
			word[vocabulary].partOfSpeech=str.substring(split1+1,split2); //[a,b)
			vocabulary++;
		}
		
		
		while(true) {
			System.out.println("請選擇模式(看中測英打1 , 看英測中打2)");
			String mode=cin.next();
			if(mode.equals("1")) {
				testEnglish(word , vocabulary);
				break;
			}
			else
				if(mode.equals("2")) {
					testChinese(word , vocabulary);
					break;
				}
				else {
					System.out.println("請再輸入一次");
				}
		}
		cout.close();
		
	}
	static void testEnglish(Word [] word , int vocabulary) {
		
		while(true) {
			int random = (int) (Math.random()*vocabulary);
			while(word[random].judge) {
				random = (int) (Math.random()*vocabulary);
			}
			System.out.println(word[random].partOfSpeech+word[random].CH);
			
			String answer=cin.next();
			if(answer.equals(word[random].ENG)) {
				System.out.println("答對了");
				word[random].judge=true;
			}
			else {
				System.out.println("答錯了");
				System.out.println(word[random].ENG);
			}
		}
		
	}
	
	static void testChinese(Word [] word , int vocabulary) {
		
		while(true) {
			int random = (int) (Math.random()*vocabulary);
			while(word[random].judge) {
				random = (int) (Math.random()*vocabulary);
			}
			System.out.println(word[random].partOfSpeech+word[random].ENG);
			
			String answer=cin.next();
			System.out.println(word[random].CH);
			word[random].judge=true;
		}
		
	}

}
class Word{
	String ENG=" ";
	String partOfSpeech=" ";
	String CH=" ";
	
	boolean judge=false;
	
}
